Leave-One-Out Cross Validation of 25 PCs

column = 25
MSE = vector(mode="double", length=column-1)
MSE_train = vector(mode="double", length=column-1)
for (j in 2:column){
  glm.pred = rep("x", length(train_pos))
  for (i in 1:length(train_pos)){
    
    glm.fit = glm(SubType~., data = df[-c(test_pos, train_pos[i]),
                                       1:j], family = binomial)
    glm.prob = predict(glm.fit, df[train_pos[i],1:j], type ="response")
    
    glm.pred[i] = ifelse(glm.prob >.5, "IDHmut-non-codel",
                         "IDHmut-codel")
  }
  glm.fit = glm(SubType~., data = df[train_pos,1:j],
                family = binomial)
  glm.prob.train = predict(glm.fit, df[train_pos,1:j], type ="response")
  
  glm.pred.train = ifelse(glm.prob.train >.5, "IDHmut-non-codel",
                       "IDHmut-codel")
  MSE_train[j-1] = mean(glm.pred.train != df$SubType[train_pos])
  MSE[j-1] <- mean(glm.pred != df$SubType[train_pos])
}

Plot the LOOCV error of the principal components

plot(MSE, type="b",pch=20,col="blue",ylab = "LOOCV Error",
     xlab = "Principal Component", cex=0.7)
points(match(min(MSE), MSE), MSE[match(min(MSE), MSE)],
       col = "red", pch=20)
text(match(min(MSE), MSE), MSE[match(min(MSE), MSE)],
     labels = match(min(MSE), MSE), pos = 3, cex = 0.7, col="red", pch =20)
